CREATE TABLE users (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  name VARCHAR(20) NOT NULL,
  password VARCHAR(20) NOT NULL,
  picture VARCHAR(255),
  firstName VARCHAR(50),
  lastName VARCHAR(50),
  email VARCHAR(100),
  dateOfBirth VARCHAR(10),
  comarca VARCHAR(50),
  role VARCHAR(10) NOT NULL DEFAULT 'user'
);

CREATE TABLE tweets (
  id INTEGER PRIMARY KEY AUTOINCREMENT,
  user_id INTEGER NOT NULL,
  parent_id INTEGER,
  title VARCHAR(255) NOT NULL,
  picture VARCHAR(255),
  textBody VARCHAR(280) NOT NULL,
  likes INTEGER NOT NULL DEFAULT 0,
  comments INTEGER NOT NULL DEFAULT 0,
  time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  is_parent BOOLEAN NOT NULL DEFAULT TRUE,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (parent_id) REFERENCES tweets(id) ON DELETE CASCADE
);

CREATE TABLE likes (
  tweet_id INTEGER NOT NULL,
  user_id INTEGER NOT NULL,
  time DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (tweet_id, user_id),
  FOREIGN KEY (tweet_id) REFERENCES tweets(id) ON DELETE CASCADE,
  FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);

CREATE TABLE follows (
  u_following INTEGER NOT NULL,
  u_followed INTEGER NOT NULL,
  followed_at DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (u_following, u_followed),
  FOREIGN KEY (u_following) REFERENCES users(id) ON DELETE CASCADE,
  FOREIGN KEY (u_followed) REFERENCES users(id) ON DELETE CASCADE
);
